function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('PMIX Pipeline (Neon)')
    .addItem('Run Full Pipeline',              'runFullPipeline')
    .addSeparator()
    .addItem('Step 1 — Build Raw MB',          'stepBuildRawMenuBreakdown')
    .addItem('Step 2 — Build Raw Modifiers',   'stepBuildRawModifiers')
    .addItem('Step 3 — Build Clean MB',        'stepBuildCleanMenuBreakdown')
    .addItem('Step 4 — Build Clean Modifiers', 'stepBuildCleanModifiers')
    .addItem('Step 5 — Build Pink Sheets',     'stepBuildPinkSheets')
    .addItem('Step 6 — Build Channel Masters', 'stepBuildChannelMasters')
    .addItem('Step 7 — Build Blended Master',  'stepBuildBlendedMaster')
    .addItem('Step 8 — Build Overall Master',  'stepBuildOverallMaster')
    .addItem('Step 9 — Populate Bikky Rates',  'stepPopulateBikky')
    .addToUi();
}

function runFullPipeline() {
  var ui   = SpreadsheetApp.getUi();
  var resp = ui.alert('PMIX Pipeline (Neon)',
    'This will rebuild all sheets from the Neon DB export tabs.\n\nContinue?',
    ui.ButtonSet.YES_NO);
  if (resp !== ui.Button.YES) return;
  var t0 = Date.now();
  try {
    toast_('Step 1/9 — Building Raw Menu Breakdown');
    stepBuildRawMenuBreakdown();
    toast_('Step 2/9 — Building Raw Modifiers');
    stepBuildRawModifiers();
    toast_('Step 3/9 — Building Clean Menu Breakdown');
    stepBuildCleanMenuBreakdown();
    toast_('Step 4/9 — Building Clean Modifiers');
    stepBuildCleanModifiers();
    toast_('Step 5/9 — Building Pink Sheets');
    stepBuildPinkSheets();
    toast_('Step 6/9 — Building Channel Masters');
    stepBuildChannelMasters();
    toast_('Step 7/9 — Building Blended Master');
    stepBuildBlendedMaster();
    toast_('Step 8/9 — Building Overall Master');
    stepBuildOverallMaster();
    toast_('Step 9/9 — Populating Bikky Rates');
    stepPopulateBikky();
    var secs = ((Date.now() - t0) / 1000).toFixed(1);
    ui.alert('Pipeline Complete', 'All 9 steps finished in ' + secs + 's.', ui.ButtonSet.OK);
  } catch (e) {
    ui.alert('Pipeline Error', e.message + '\n\n' + e.stack, ui.ButtonSet.OK);
  }
}


// ============================================================
// HELPERS
// ============================================================
function toast_(msg) {
  SpreadsheetApp.getActiveSpreadsheet().toast(msg, 'PMIX Pipeline', 60);
}

function getSheet_(name) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(name);
  if (!sh) throw new Error('Sheet not found: "' + name + '"');
  return sh;
}

function getOrCreate_(name) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(name);
  if (!sh) sh = ss.insertSheet(name);
  return sh;
}

function clearAndCreate_(name) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(name);
  if (sh) { sh.clearContents(); sh.clearFormats(); }
  else sh = ss.insertSheet(name);
  return sh;
}

function getDataMap_(sheetName) {
  // Returns { headers: [...], rows: [[...], ...] } with all values from sheet
  var sh   = getSheet_(sheetName);
  var data = sh.getDataRange().getValues();
  return { headers: data[0], rows: data.slice(1) };
}

function getCat_(name, grp) {
  if (ITEM_CATEGORY_OVERRIDE[name]) return ITEM_CATEGORY_OVERRIDE[name];
  return GRP_TO_CATEGORY[grp] || '';
}
function getSubCat_(name, grp) {
  if (ITEM_SUBCATEGORY_OVERRIDE[name]) return ITEM_SUBCATEGORY_OVERRIDE[name];
  if (ITEM_SUBCATEGORY[name])          return ITEM_SUBCATEGORY[name];
  return GRP_TO_SUBCATEGORY[grp] || '';
}
function menuEng_(mHigh, mmHigh) {
  if (mHigh && mmHigh)  return 'Star';
  if (mHigh && !mmHigh) return 'Puzzle';
  if (!mHigh && mmHigh) return 'Plow Horse';
  return 'Dog';
}

function applyHeaderStyle_(ws, row, ncols) {
  var range = ws.getRange(row, 1, 1, ncols);
  range.setFontWeight('bold')
       .setBackground('#4B0082')
       .setFontColor('#FFFFFF')
       .setFontFamily('Arial')
       .setFontSize(9)
       .setHorizontalAlignment('center')
       .setWrap(true);
}

function applyGrandTotalStyle_(ws, row, ncols) {
  var range = ws.getRange(row, 1, 1, ncols);
  range.setFontWeight('bold')
       .setBackground('#7B2FBE')
       .setFontColor('#FFFFFF')
       .setFontFamily('Arial')
       .setFontSize(9);
}

function writeSheetData_(ws, headers, dataRows, grandTotalRow, thresholdRows) {
  // headers: array of strings
  // dataRows: array of arrays
  // grandTotalRow: optional single array (appended after data)
  // thresholdRows: optional array of arrays (appended after grand total)
  var nc   = headers.length;
  var allRows = [headers].concat(dataRows);
  if (grandTotalRow) allRows.push(grandTotalRow);
  if (thresholdRows) thresholdRows.forEach(function(r) { allRows.push(r); });

  ws.clearContents();
  ws.clearFormats();
  ws.getRange(1, 1, allRows.length, nc).setValues(allRows);

  applyHeaderStyle_(ws, 1, nc);

  // Alternating row colors for data
  for (var r = 0; r < dataRows.length; r++) {
    var bg = (r % 2 === 0) ? '#FFFFFF' : '#EDE7F6';
    ws.getRange(r + 2, 1, 1, nc).setBackground(bg).setFontFamily('Arial').setFontSize(9);
  }

  // Grand total
  if (grandTotalRow) applyGrandTotalStyle_(ws, dataRows.length + 2, nc);

  // Threshold rows
  if (thresholdRows) {
    var startRow = dataRows.length + (grandTotalRow ? 3 : 2);
    ws.getRange(startRow, 1, thresholdRows.length, nc)
      .setBackground('#F3E5F5')
      .setFontStyle('italic')
      .setFontSize(8)
      .setFontFamily('Arial');
  }

  ws.setFrozenRows(1);
}

// STEP 1 — BUILD RAW MENU BREAKDOWN
// Aggregates fact_order_lines into the same format as Toast's
// Menu Breakdown CSV export: one row per (Menu, Menu Group, Item).
// ============================================================
function stepBuildRawMenuBreakdown() {
  var ordData = getDataMap_(SH.FACT_ORDERS);
  var hdrs    = ordData.headers;

  // Column indices (0-based)
  var C = {
    channel:    hdrs.indexOf('channel_code'),
    canonical:  hdrs.indexOf('canonical_name'),
    menuName:   hdrs.indexOf('menu_name'),
    menuGroup:  hdrs.indexOf('menu_group'),
    qty:        hdrs.indexOf('quantity'),
    lineTotal:  hdrs.indexOf('line_total'),
    preDisc:    hdrs.indexOf('pre_discount'),
    isVoided:   hdrs.indexOf('is_voided'),
  };

  // Aggregate: key = channel + '|' + menuGroup + '|' + canonicalName
  var agg = {};  // key -> { menu, grp, name, qty, gross, net }

  ordData.rows.forEach(function(row) {
    if (row[C.isVoided] === true || row[C.isVoided] === 't' || row[C.isVoided] === 'true') return;
    var chCode = String(row[C.channel] || '').trim();
    if (EXCLUDED_CHANNELS[chCode]) return;

    var menuName = String(row[C.menuName] || '').trim();
    if (!menuName || menuName === '') return;  // skip rows with no menu (order notes etc.)
    if (EXCLUDED_GROUPS[String(row[C.menuGroup] || '').trim()]) return;

    var channel = CHANNEL_MAP[chCode];
    if (!channel) return;

    var grp  = String(row[C.menuGroup]  || '').trim();
    var name = String(row[C.canonical]  || '').trim();
    var qty  = Number(row[C.qty])  || 0;
    var net  = Number(row[C.lineTotal]) || 0;
    var gross= Number(row[C.preDisc])   || 0;
    if (!name) return;

    var key = channel + '||' + grp + '||' + name;
    if (!agg[key]) agg[key] = { channel: channel, grp: grp, name: name, qty: 0, gross: 0, net: 0 };
    agg[key].qty   += qty;
    agg[key].gross += gross;
    agg[key].net   += net;
  });

  var ws = clearAndCreate_(SH.RAW_MB);
  var headers = ['Menu','Menu Group','Item Name','Modifier',
                 'Avg Price','Quantity','Gross Sales','Discount Amount','Net Sales'];
  var rows = [];
  Object.keys(agg).forEach(function(k) {
    var d = agg[k];
    var avgPrice = d.qty ? d.net / d.qty : 0;
    rows.push([d.channel, d.grp, d.name, '',
               avgPrice, d.qty, d.gross, d.gross - d.net, d.net]);
  });

  // Sort: channel, then menu group, then name
  rows.sort(function(a, b) {
    if (a[0] < b[0]) return -1; if (a[0] > b[0]) return 1;
    if (a[1] < b[1]) return -1; if (a[1] > b[1]) return 1;
    if (a[2] < b[2]) return -1; if (a[2] > b[2]) return 1;
    return 0;
  });

  ws.getRange(1, 1, 1, headers.length).setValues([headers]);
  applyHeaderStyle_(ws, 1, headers.length);
  if (rows.length) {
    ws.getRange(2, 1, rows.length, headers.length).setValues(rows);
    ws.getRange(2, 5, rows.length, 1).setNumberFormat('"$"#,##0.00');
    ws.getRange(2, 7, rows.length, 1).setNumberFormat('"$"#,##0.00');
    ws.getRange(2, 8, rows.length, 1).setNumberFormat('"$"#,##0.00');
    ws.getRange(2, 9, rows.length, 1).setNumberFormat('"$"#,##0.00');
  }
  ws.setFrozenRows(1);
  ws.setColumnWidth(1, 120); ws.setColumnWidth(2, 160);
  ws.setColumnWidth(3, 240); ws.setColumnWidth(6, 80);
  toast_('Raw Menu Breakdown: ' + rows.length + ' rows');
}

// STEP 2 — BUILD RAW MODIFIERS
// Joins fact_modifiers to fact_order_lines on parent_selection
// to recover the parent item name and menu group.
// ============================================================
function stepBuildRawModifiers() {
  var ordData = getDataMap_(SH.FACT_ORDERS);
  var modData = getDataMap_(SH.FACT_MODS);

  // Index order lines by selection_guid -> { canonical, menuGroup, channelCode }
  var oHdrs = ordData.headers;
  var oC = {
    guid:      oHdrs.indexOf('selection_guid'),
    canonical: oHdrs.indexOf('canonical_name'),
    menuGroup: oHdrs.indexOf('menu_group'),
    channel:   oHdrs.indexOf('channel_code'),
    isVoided:  oHdrs.indexOf('is_voided'),
  };
  var orderIndex = {};
  ordData.rows.forEach(function(row) {
    if (row[oC.isVoided] === true || row[oC.isVoided] === 't') return;
    var ch = String(row[oC.channel] || '').trim();
    if (EXCLUDED_CHANNELS[ch]) return;
    var guid = String(row[oC.guid] || '').trim();
    if (!guid) return;
    orderIndex[guid] = {
      parent:  String(row[oC.canonical] || '').trim(),
      grp:     String(row[oC.menuGroup] || '').trim(),
      channel: ch,
    };
  });

  var mHdrs = modData.headers;
  var mC = {
    guid:      mHdrs.indexOf('modifier_guid'),
    parentSel: mHdrs.indexOf('parent_selection'),
    canonical: mHdrs.indexOf('canonical_name'),
    qty:       mHdrs.indexOf('quantity'),
    price:     mHdrs.indexOf('price'),
    isVoided:  mHdrs.indexOf('is_voided'),
  };

  // Aggregate: (modifier, parentItem, menuGroup) -> { qty, netAmount }
  var agg = {};
  modData.rows.forEach(function(row) {
    if (row[mC.isVoided] === true || row[mC.isVoided] === 't') return;
    var parentSel = String(row[mC.parentSel] || '').trim();
    var ord = orderIndex[parentSel];
    if (!ord) return;
    if (EXCLUDED_GROUPS[ord.grp]) return;

    var modName = String(row[mC.canonical] || '').trim();
    var qty     = Number(row[mC.qty])   || 0;
    var price   = Number(row[mC.price]) || 0;
    var netAmt  = qty * price;

    var key = modName + '||' + ord.parent + '||' + ord.grp;
    if (!agg[key]) agg[key] = { mod: modName, parent: ord.parent, grp: ord.grp, qty: 0, net: 0 };
    agg[key].qty += qty;
    agg[key].net += netAmt;
  });

  var ws = clearAndCreate_(SH.RAW_MOD);
  var headers = ['Modifiers','Parent Item','Menu Group','Qty','Net Amount'];
  var rows = [];
  Object.keys(agg).forEach(function(k) {
    var d = agg[k];
    rows.push([d.mod, d.parent, d.grp, d.qty, d.net]);
  });
  rows.sort(function(a, b) {
    if (a[1] < b[1]) return -1; if (a[1] > b[1]) return 1;
    return 0;
  });

  ws.getRange(1, 1, 1, headers.length).setValues([headers]);
  applyHeaderStyle_(ws, 1, headers.length);
  if (rows.length) {
    ws.getRange(2, 1, rows.length, headers.length).setValues(rows);
    ws.getRange(2, 5, rows.length, 1).setNumberFormat('"$"#,##0.00');
  }
  ws.setFrozenRows(1);
  ws.setColumnWidth(1, 220); ws.setColumnWidth(2, 200); ws.setColumnWidth(3, 180);
  toast_('Raw Modifiers: ' + rows.length + ' rows');
}

// STEP 3 — BUILD CLEAN MENU BREAKDOWN
// Applies channel mapping, BYO name fixes, and item cost lookup
// from analytics.r365_item_cost.
// ============================================================
function stepBuildCleanMenuBreakdown() {
  // Build item cost map from r365_item_cost
  var icData = getDataMap_(SH.R365_ITEM);
  var icH = icData.headers;
  var icC = {
    menu:    icH.indexOf('menu'),
    iname:   icH.indexOf('item_name'),
    iupd:    icH.indexOf('item_name_updated'),
    grp:     icH.indexOf('menu_group'),
    cat1:    icH.indexOf('category_1'),
    cat2:    icH.indexOf('category_2'),
    cost:    icH.indexOf('avg_cost'),
  };
  // Map: (channel, rawName) -> { cost, cat1, cat2, cleanName }
  var MENU_TO_CHANNEL = {
    'DELIVERY':'DELIVERY','3PD OPEN MARKUP':'DELIVERY',
    'APP':'LOYALTY','FOOD - TOAST ONLINE ORDERING':'LOYALTY',
    'FOOD - IN HOUSE':'IN HOUSE','DRINKS - IN HOUSE':'IN HOUSE',
  };
  var costMap = {};   // key = channel+'||'+name -> { cost, cleanName, cat1, cat2 }
  icData.rows.forEach(function(row) {
    var menu  = String(row[icC.menu]  || '').trim();
    var iname = String(row[icC.iname] || '').trim();
    var iupd  = String(row[icC.iupd]  || '').trim();
    var cat1  = String(row[icC.cat1]  || '').trim();
    var cat2  = String(row[icC.cat2]  || '').trim();
    var cost  = row[icC.cost];
    var ch    = MENU_TO_CHANNEL[menu] || menu;
    var entry = { cost: (typeof cost === 'number' ? cost : null), cleanName: iupd || iname, cat1: cat1, cat2: cat2 };
    if (iname) costMap[ch + '||' + iname] = entry;
    if (iupd && iupd !== iname) costMap[ch + '||' + iupd] = entry;
  });

  // Read raw MB
  var rawData = getSheet_(SH.RAW_MB).getDataRange().getValues();
  var ws = clearAndCreate_(SH.CLEAN_MB);
  var headers = [
    'Menu','Menu Group','Item Name','Modifier','Avg Price','Quantity',
    'Gross Sales','Discount Amount','Net Sales',
    'Item name_updated','Category 1','Category 2','Avg Cost',
    'Cost Margin','Profit %','Total Cost','Total Cost Margin','Profit % (Final)',
    'Menu (For Kutlerri)'
  ];

  var rows = [];
  for (var r = 1; r < rawData.length; r++) {
    var channel = String(rawData[r][0] || '').trim();
    var grp     = String(rawData[r][1] || '').trim();
    var iname   = String(rawData[r][2] || '').trim();
    var avgPrice= Number(rawData[r][4]) || 0;
    var qty     = Number(rawData[r][5]) || 0;
    var gross   = Number(rawData[r][6]) || 0;
    var disc    = Number(rawData[r][7]) || 0;
    var net     = Number(rawData[r][8]) || 0;
    if (!channel || !iname) continue;

    var cleanName = BYO_NAME_MAP[iname] || iname;

    var entry = costMap[channel + '||' + iname]
             || costMap[channel + '||' + cleanName]
             || costMap['DELIVERY||' + cleanName]
             || {};
    var avgCost   = (entry.cost !== undefined && entry.cost !== null) ? entry.cost : null;
    var cat1      = entry.cat1 || getCat_(cleanName, grp);
    var cat2      = entry.cat2 || getSubCat_(cleanName, grp);
    var finalClean= entry.cleanName || cleanName;

    var costMargin  = (avgCost !== null) ? avgPrice - avgCost : '';
    var profitPct   = (avgCost !== null && avgPrice) ? (avgPrice - avgCost) / avgPrice : '';
    var totalCost   = (avgCost !== null) ? avgCost * qty : '';
    var totalCostMg = (costMargin !== '') ? costMargin * qty : '';
    var profitFinal = (totalCostMg !== '' && net) ? totalCostMg / net : '';

    rows.push([
      channel, grp, iname, '',
      avgPrice, qty, gross, disc, net,
      finalClean, cat1, cat2,
      (avgCost !== null) ? avgCost : '',
      costMargin, profitPct, totalCost, totalCostMg, profitFinal,
      channel
    ]);
  }

  ws.getRange(1, 1, 1, headers.length).setValues([headers]);
  applyHeaderStyle_(ws, 1, headers.length);
  if (rows.length) {
    ws.getRange(2, 1, rows.length, headers.length).setValues(rows);
    var dr = rows.length;
    ws.getRange(2,  5, dr, 1).setNumberFormat('"$"#,##0.00');
    ws.getRange(2,  9, dr, 1).setNumberFormat('"$"#,##0.00');
    ws.getRange(2, 13, dr, 1).setNumberFormat('"$"#,##0.00');
    ws.getRange(2, 15, dr, 1).setNumberFormat('0.00%');
    ws.getRange(2, 18, dr, 1).setNumberFormat('0.00%');
  }
  ws.setFrozenRows(1);
  toast_('Clean Menu Breakdown: ' + rows.length + ' rows');
}


// ============================================================
// STEP 4 — BUILD CLEAN MODIFIERS
// Joins Raw Modifiers to analytics.parent_item (for item_type)
// and analytics.lookup_modifier_type (for modifier_type),
// with channel-aware IH vs Online item type resolution.
// ============================================================
function stepBuildCleanModifiers() {
  // Build parent_item type map (item_type per parent_item name)
  var piData = getDataMap_(SH.PARENT_ITEM);
  var piH = piData.headers;
  var piC = { parent: piH.indexOf('parent_item'), itype: piH.indexOf('item_type') };
  var itemTypeOnline = {};  // canonical parent -> online item_type
  var itemTypeIH     = {};  // canonical parent (short form) -> IH item_type
  piData.rows.forEach(function(row) {
    var p = String(row[piC.parent] || '').trim();
    var t = String(row[piC.itype]  || '').trim();
    if (!p || !t) return;
    if (t.indexOf('In House') !== -1) {
      itemTypeIH[p] = t;
      // Also register the short base name (strip ' - In House')
      var base = p.replace(/ - In House$/i, '');
      if (!itemTypeIH[base]) itemTypeIH[base] = t;
    } else {
      itemTypeOnline[p] = t;
    }
  });
  // Merge IH_PARENT_TYPE overrides
  Object.keys(IH_PARENT_TYPE).forEach(function(k) {
    if (!itemTypeIH[k]) itemTypeIH[k] = IH_PARENT_TYPE[k];
  });

  // Build modifier type lookup map
  var lmtData = getDataMap_(SH.LKP_MODTYPE);
  var lmtH = lmtData.headers;
  var lmtC = { mod: lmtH.indexOf('modifier_name'), itype: lmtH.indexOf('item_type'), mtype: lmtH.indexOf('modifier_type') };
  var modTypeMap = {};  // modName + '|' + itemType -> modifierType
  lmtData.rows.forEach(function(row) {
    var mod   = String(row[lmtC.mod]   || '').trim();
    var itype = String(row[lmtC.itype] || '').trim();
    var mtype = String(row[lmtC.mtype] || '').trim();
    if (mod && itype && mtype) modTypeMap[mod + '|' + itype] = mtype;
  });

  // Build R365 modifier name map: raw recipe name -> clean_name
  var mcData = getDataMap_(SH.R365_MOD);
  var mcH = mcData.headers;
  var mcC = { rname: mcH.indexOf('recipe_name'), cname: mcH.indexOf('clean_name') };
  var r365NameMap = {};
  mcData.rows.forEach(function(row) {
    var rn = String(row[mcC.rname] || '').trim();
    var cn = String(row[mcC.cname] || '').trim();
    if (rn.indexOf('MI ') === 0 && cn && !r365NameMap[rn.slice(3)]) {
      r365NameMap[rn.slice(3)] = cn;
    }
  });

  // Build index of order line channel by selection_guid
  var ordData = getDataMap_(SH.FACT_ORDERS);
  var oH = ordData.headers;
  var oC = {
    guid:     oH.indexOf('selection_guid'),
    channel:  oH.indexOf('channel_code'),
    isVoided: oH.indexOf('is_voided'),
  };
  var orderChannel = {};
  ordData.rows.forEach(function(row) {
    if (row[oC.isVoided] === true || row[oC.isVoided] === 't') return;
    var ch = String(row[oC.channel] || '').trim();
    if (!EXCLUDED_CHANNELS[ch]) {
      orderChannel[String(row[oC.guid] || '').trim()] = ch;
    }
  });

  // Read raw modifiers and re-join with channel info from fact_modifiers
  var modFact = getDataMap_(SH.FACT_MODS);
  var mfH = modFact.headers;
  var mfC = {
    parentSel: mfH.indexOf('parent_selection'),
    canonical: mfH.indexOf('canonical_name'),
    qty:       mfH.indexOf('quantity'),
    price:     mfH.indexOf('price'),
    isVoided:  mfH.indexOf('is_voided'),
  };

  // Index raw_mod sheet for parent item + grp
  var rawModData = getSheet_(SH.RAW_MOD).getDataRange().getValues();
  // Build map: modName + '||' + parentName -> grp
  var rawModGrp = {};
  for (var r = 1; r < rawModData.length; r++) {
    var key = String(rawModData[r][0]||'').trim() + '||' + String(rawModData[r][1]||'').trim();
    if (!rawModGrp[key]) rawModGrp[key] = String(rawModData[r][2]||'').trim();
  }

  // Aggregate: (cleanMod, parentItem, itemType) -> { qty, net, mtype }
  // Use fact_modifiers to get channel, then resolve item_type channel-aware
  var agg = {};  // key = cleanMod + '||' + parent + '||' + itype

  modFact.rows.forEach(function(row) {
    if (row[mfC.isVoided] === true || row[mfC.isVoided] === 't') return;
    var parentSel = String(row[mfC.parentSel] || '').trim();
    var chCode    = orderChannel[parentSel];
    if (!chCode) return;

    // We need parent canonical from the original order line index
    // Re-use orderChannel map which has channel but not parent name
    // So we read raw_mod for the aggregated (mod, parent) pair
    // Actually we need to re-join. Simpler: process from raw_mod sheet + channel from fact_modifiers.
    // Skip — handled via raw_mod loop below
  });

  // Process from raw_mod sheet directly
  // For each (modifier, parentItem, menuGroup) row in raw_mod,
  // we need to know channel. We can infer: if parentItem has IH item type,
  // the IH data came from IN_HOUSE channel. But some items appear in both.
  // Solution: re-aggregate fact_modifiers with both channel and parent.

  var modByChannelParent = {};  // parentName + '|' + chCode + '|' + modName -> { qty, net, grp }
  var ordLineIndex = {};  // selection_guid -> { parent, grp, channel }
  ordData.rows.forEach(function(row) {
    if (row[oC.isVoided] === true || row[oC.isVoided] === 't') return;
    var ch = String(row[oC.channel] || '').trim();
    if (EXCLUDED_CHANNELS[ch]) return;
    var guid = String(row[oC.guid] || '').trim();
    if (!guid) return;
    // Re-index with parent canonical
    var parentH = ordData.headers;
    ordLineIndex[guid] = {
      parent:  String(row[parentH.indexOf('canonical_name')] || '').trim(),
      grp:     String(row[parentH.indexOf('menu_group')]     || '').trim(),
      channel: ch,
    };
  });

  modFact.rows.forEach(function(row) {
    if (row[mfC.isVoided] === true || row[mfC.isVoided] === 't') return;
    var parentSel = String(row[mfC.parentSel] || '').trim();
    var ord = ordLineIndex[parentSel];
    if (!ord) return;
    if (EXCLUDED_GROUPS[ord.grp]) return;

    var modName = String(row[mfC.canonical] || '').trim();
    var qty     = Number(row[mfC.qty])   || 0;
    var net     = Number(row[mfC.price]) * qty || 0;
    var chCode  = ord.channel;

    var key = ord.parent + '||' + chCode + '||' + modName;
    if (!modByChannelParent[key]) {
      modByChannelParent[key] = { mod: modName, parent: ord.parent, grp: ord.grp, channel: chCode, qty: 0, net: 0 };
    }
    modByChannelParent[key].qty += qty;
    modByChannelParent[key].net += net;
  });

  // Now build clean modifiers output
  var ws = clearAndCreate_(SH.CLEAN_MOD);
  var headers = ['modifier','Parent item','Item Type','','Menu Group','Qty','Net Amount','Concat','Modifier type_new'];
  var rows = [];

  Object.keys(modByChannelParent).forEach(function(k) {
    var d = modByChannelParent[k];
    var isIH = (d.channel === 'IN_HOUSE');

    // Resolve item type
    var itype;
    if (isIH) {
      itype = itemTypeIH[d.parent] || itemTypeOnline[d.parent] || '';
    } else {
      itype = itemTypeOnline[d.parent] || '';
    }

    // Clean modifier name
    var cleanMod = r365NameMap[d.mod] || d.mod;

    // Modifier type lookup
    var mtype = modTypeMap[cleanMod + '|' + itype]
             || modTypeMap[d.mod + '|' + itype]
             || '';
    if (!mtype && cleanMod.indexOf('Organic ') === 0) {
      mtype = modTypeMap[cleanMod.slice(8) + '|' + itype] || '';
    }
    if (!mtype) mtype = MODTYPE_OVERRIDES[cleanMod] || MODTYPE_OVERRIDES[d.mod] || '';

    var concat = cleanMod + itype;
    rows.push([cleanMod, d.parent, itype, '', d.grp, d.qty, d.net, concat, mtype]);
  });

  rows.sort(function(a, b) {
    if (a[1] < b[1]) return -1; if (a[1] > b[1]) return 1;
    if (a[0] < b[0]) return -1; if (a[0] > b[0]) return 1;
    return 0;
  });

  ws.getRange(1, 1, 1, headers.length).setValues([headers]);
  applyHeaderStyle_(ws, 1, headers.length);
  if (rows.length) {
    ws.getRange(2, 1, rows.length, headers.length).setValues(rows);
    ws.getRange(2, 7, rows.length, 1).setNumberFormat('"$"#,##0.00');
  }
  ws.setFrozenRows(1);
  ws.setColumnWidth(1, 200); ws.setColumnWidth(2, 180); ws.setColumnWidth(3, 200);
  ws.setColumnWidth(5, 160); ws.setColumnWidth(9, 160);
  toast_('Clean Modifiers: ' + rows.length + ' rows');
}

// STEP 5 — BUILD PINK SHEETS
// Builds one cost breakdown sheet per item (online and IH variant)
// mirroring the GAS pink sheet format exactly.
// ============================================================
function stepBuildPinkSheets() {
  var mcData = getDataMap_(SH.R365_MOD);
  var mcH = mcData.headers;
  var mcC = { rname: mcH.indexOf('recipe_name'), cname: mcH.indexOf('clean_name'), cost: mcH.indexOf('cost_per_portion') };

  // Build modifier cost map: norm(cleanName) -> cost_per_portion
  var modCostMap = {};
  mcData.rows.forEach(function(row) {
    var rn   = String(row[mcC.rname] || '').trim();
    var cn   = String(row[mcC.cname] || '').trim();
    var cost = row[mcC.cost];
    if (rn.indexOf('MI ') === 0 && cn && typeof cost === 'number' && cost > 0) {
      var k = normMod_(cn);
      if (!(k in modCostMap) || cost > modCostMap[k]) modCostMap[k] = cost;
    }
  });

  // Build base item cost map: (channel, itemName) -> cost
  var icData = getDataMap_(SH.R365_ITEM);
  var icH = icData.headers;
  var icC2 = { menu: icH.indexOf('menu'), iname: icH.indexOf('item_name'), iupd: icH.indexOf('item_name_updated'), cost: icH.indexOf('avg_cost') };
  var MENU_TO_CH = {
    'DELIVERY':'DELIVERY','3PD OPEN MARKUP':'DELIVERY',
    'APP':'LOYALTY','FOOD - IN HOUSE':'IN HOUSE','DRINKS - IN HOUSE':'IN HOUSE',
  };
  var itemCostMap = {};
  icData.rows.forEach(function(row) {
    var menu  = String(row[icC2.menu]  || '').trim();
    var iname = String(row[icC2.iname] || '').trim();
    var iupd  = String(row[icC2.iupd]  || '').trim();
    var cost  = row[icC2.cost];
    var ch    = MENU_TO_CH[menu] || menu;
    if (iname && typeof cost === 'number') {
      itemCostMap[ch + '||' + iname] = cost;
      if (iupd && iupd !== iname) itemCostMap[ch + '||' + iupd] = cost;
    }
  });

  // Build modifier data index: (parentItem, itemType) -> { modType -> { modName -> qty } }
  var cmData = getSheet_(SH.CLEAN_MOD).getDataRange().getValues();
  var modData = {};  // key = parentItem + '||' + itemType -> { mtype -> { mod -> qty } }
  for (var r = 1; r < cmData.length; r++) {
    var cleanMod  = String(cmData[r][0] || '').trim();
    var parentItm = String(cmData[r][1] || '').trim();
    var itype     = String(cmData[r][2] || '').trim();
    var qty       = Number(cmData[r][5]) || 0;
    var mtype     = String(cmData[r][8] || '').trim();
    if (!cleanMod || !parentItm || qty === 0 || !mtype) continue;
    if (isCateringMod_(cleanMod) || isCateringMod_(mtype)) continue;
    var key = parentItm + '||' + itype;
    if (!modData[key]) modData[key] = {};
    if (!modData[key][mtype]) modData[key][mtype] = {};
    modData[key][mtype][cleanMod] = (modData[key][mtype][cleanMod] || 0) + qty;
  }

  // Build each pink sheet
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var count = 0;
  PINK_ITEMS.forEach(function(def) {
    var itemName = def[0], rawOn = def[1], rawIH = def[2], typeOn = def[3], typeIH = def[4];

    // Online sheet
    if (rawOn && typeOn) {
      var sname = itemName.length > 31 ? itemName.slice(0, 31) : itemName;
      var baseCost = getBaseCost_(itemCostMap, itemName, 'DELIVERY');
      writePinkSheet_(ss, sname, itemName, rawOn, typeOn, 'THANX + 3PD', modData, modCostMap, baseCost);
      count++;
    }
    // IH sheet
    if (rawIH && typeIH) {
      var snameIH = (itemName + '(in house)').length > 31 ? (itemName + '(in house)').slice(0, 31) : itemName + '(in house)';
      var baseCostIH = getBaseCost_(itemCostMap, itemName, 'IN HOUSE');
      writePinkSheet_(ss, snameIH, itemName, rawIH, typeIH, 'IN HOUSE', modData, modCostMap, baseCostIH);
      count++;
    }
  });
  toast_('Pink Sheets: ' + count + ' built');
}

function normMod_(s) {
  return String(s || '').toLowerCase().trim();
}

function isCateringMod_(s) {
  var n = String(s || '').toLowerCase();
  return n.indexOf('catering') !== -1 || n.indexOf('zerocater') !== -1 ||
         n.indexOf('sharebite') !== -1 || n.indexOf('ezcater') !== -1 ||
         n.indexOf('hungry') !== -1;
}

function getBaseCost_(itemCostMap, itemName, channel) {
  var base = itemName.endsWith(' - In House') ? itemName.slice(0, -11) : itemName;
  var isIH = (channel === 'IN HOUSE');
  if (isIH) {
    var IH_VARIANTS = {
      'Kids Meal': 'Kids BYO - In House',
      'BYO Indian Burrito': 'Burrito - In House',
      'BYO Grain Bowl': 'BYO Grain Bowl - In House',
      'BYO Salad Bowl': 'BYO Salad Bowl - In House',
      'BYO Greens + Grains Bowl': 'BYO Greens + Grains Bowl - In House',
    };
    var varKey = IH_VARIANTS[base];
    if (varKey && itemCostMap['IN HOUSE||' + varKey] !== undefined) return itemCostMap['IN HOUSE||' + varKey];
    if (itemCostMap['IN HOUSE||' + base + ' - In House'] !== undefined) return itemCostMap['IN HOUSE||' + base + ' - In House'];
    if (itemCostMap['IN HOUSE||' + base] !== undefined) return itemCostMap['IN HOUSE||' + base];
  }
  if (itemCostMap['DELIVERY||' + base] !== undefined) return itemCostMap['DELIVERY||' + base];
  if (itemCostMap['LOYALTY||' + base]  !== undefined) return itemCostMap['LOYALTY||' + base];
  if (itemCostMap['IN HOUSE||' + base] !== undefined) return itemCostMap['IN HOUSE||' + base];
  return 0;
}

function getModCost_(name, modCostMap) {
  if (!name) return null;
  var low = normMod_(name);
  if (low.indexOf('skip') === 0 || low === 'grand total' || low.indexOf('no ') === 0) return 0;
  if (modCostMap[low] !== undefined) return modCostMap[low];
  // Sauce aliases
  var SAUCE_ALIASES = {
    'tomato garlic (butter masala)': 'tomato garlic sauce',
    'tikka masala':                  'tikka masala sauce',
    'tamarind chili (spicy)':        'tamarind chili sauce',
    'peanut sesame':                 'peanut sesame sauce',
    'coconut ginger':                'coconut ginger sauce',
  };
  if (SAUCE_ALIASES[low] && modCostMap[SAUCE_ALIASES[low]] !== undefined) return modCostMap[SAUCE_ALIASES[low]];
  // Extra X → X
  if (low.indexOf('extra ') === 0) {
    var base = low.slice(6);
    if (modCostMap[base] !== undefined) return modCostMap[base];
  }
  // 1/2 X → X/2
  if (low.indexOf('1/2 ') === 0) {
    var base2 = low.slice(4).replace(/^and /, '');
    if (modCostMap[base2] !== undefined) return modCostMap[base2] / 2;
  }
  // X - Side → X
  if (low.slice(-6) === ' - side') {
    var base3 = low.slice(0, -7);
    if (modCostMap[base3] !== undefined) return modCostMap[base3];
  }
  if (low === 'tandoori paneer' && modCostMap[normMod_('Organic Tandoori Paneer')] !== undefined)
    return modCostMap[normMod_('Organic Tandoori Paneer')];
  if (low === 'romaine' && modCostMap[normMod_('Shredded Romaine')] !== undefined)
    return modCostMap[normMod_('Shredded Romaine')];
  if (low.indexOf('spicy mango chutney') !== -1) return 0.1777;
  return null;
}

function subWeightedAvg_(subMods, modCostMap) {
  var total = 0;
  Object.keys(subMods).forEach(function(m) { total += subMods[m]; });
  if (!total) return 0;
  var costSum = 0;
  Object.keys(subMods).forEach(function(m) {
    costSum += (getModCost_(m, modCostMap) || 0) * subMods[m];
  });
  return costSum / total;
}

function writePinkSheet_(ss, sheetName, itemName, rawParent, itemType, channelLabel, modData, modCostMap, baseCost) {
  var ws = ss.getSheetByName(sheetName);
  if (ws) { ws.clearContents(); ws.clearFormats(); }
  else ws = ss.insertSheet(sheetName);

  // Get sections
  var key = rawParent + '||' + itemType;
  var rawSections = modData[key] || {};
  var sections = {};
  Object.keys(rawSections).forEach(function(k) {
    sections[k] = {};
    Object.keys(rawSections[k]).forEach(function(m) { sections[k][m] = rawSections[k][m]; });
  });

  // Merge Side-Sauce -> Sauce
  if (sections['Side - Sauce']) {
    if (!sections['Sauce']) sections['Sauce'] = {};
    Object.keys(sections['Side - Sauce']).forEach(function(m) {
      sections['Sauce'][m] = (sections['Sauce'][m] || 0) + sections['Side - Sauce'][m];
    });
    delete sections['Side - Sauce'];
  }
  // Merge Side-Chutney -> Chutney (Neon uses 'Chutney + Dressing')
  var chutneyKey = 'Side - Chutney + Dressing';
  if (sections[chutneyKey]) {
    if (!sections['Chutney + Dressing']) sections['Chutney + Dressing'] = {};
    Object.keys(sections[chutneyKey]).forEach(function(m) {
      sections['Chutney + Dressing'][m] = (sections['Chutney + Dressing'][m] || 0) + sections[chutneyKey][m];
    });
    delete sections[chutneyKey];
  }
  // Merge Make It Meal
  var mealMods = {};
  ['Side','Drink','Sweet'].forEach(function(t) {
    if (sections[t]) {
      Object.keys(sections[t]).forEach(function(m) {
        mealMods[m] = (mealMods[m] || 0) + sections[t][m];
      });
      delete sections[t];
    }
  });

  var halfBase = sections['1/2 Base'] || {};
  var halfMain = sections['1/2 Main'] || {};
  var baseIsEmpty = !sections['Base'] || Object.keys(sections['Base']).length === 0;
  var useHalfAsPrimary = baseIsEmpty && sections['1/2 Base'] && Object.keys(sections['1/2 Base']).length > 0;

  // Render order
  var SECTION_ORDER = [
    { key: useHalfAsPrimary ? '1/2 Base' : 'Base', label: 'BASES',
      sub: useHalfAsPrimary ? {} : halfBase, exclCost: false },
    { key: 'Main',         label: 'MAIN',                sub: halfMain, exclCost: false },
    { key: 'Plate - Main', label: 'MAIN',                sub: {},       exclCost: true  },
    { key: 'Kids - Main',  label: 'MAINS',               sub: {},       exclCost: false },
    { key: 'Extra Main',   label: 'EXTRA MAIN',          sub: {},       exclCost: false },
    { key: 'Sauce',        label: 'SAUCE',               sub: {},       exclCost: false },
    { key: 'Veggie',       label: 'VEGGIES',             sub: {},       exclCost: false },
    { key: 'Extra Veggie', label: 'EXTRA VEGGIES',       sub: {},       exclCost: false },
    { key: 'Topping',      label: 'TOPPINGS',            sub: {},       exclCost: false },
    { key: 'Chutney + Dressing', label: 'CHUTNEY AND DRESSINGS', sub: {}, exclCost: false },
    { key: '__MEAL__',     label: 'MAKE IT MEAL',        sub: {},       exclCost: false },
  ];

  var toRender = [];
  SECTION_ORDER.forEach(function(s) {
    if (s.key === '__MEAL__') {
      if (Object.keys(mealMods).length > 0) toRender.push({ key: '__MEAL__', label: s.label, mods: mealMods, sub: {}, exclCost: false });
    } else if (sections[s.key] && Object.keys(sections[s.key]).length > 0) {
      toRender.push({ key: s.key, label: s.label, mods: sections[s.key], sub: s.sub, exclCost: s.exclCost });
    }
  });

  var NUM_COLS = 13;
  var allRows = [];
  var allStyles = [];  // { bg, bold }

  function pushRow(vals, bold, bg) {
    var r = [];
    for (var i = 0; i < NUM_COLS; i++) r.push(i < vals.length ? (vals[i] !== undefined ? vals[i] : '') : '');
    allRows.push(r);
    allStyles.push({ bold: !!bold, bg: bg || null });
  }
  function spacer() { pushRow([], false, null); }

  var totalModCost = 0;

  toRender.forEach(function(sec) {
    var label = sec.label; var mods = sec.mods; var sub = sec.sub; var exclCost = sec.exclCost;
    var hasSubTab = Object.keys(sub).length > 0;
    var subArr = Object.keys(sub).map(function(m) { return [m, sub[m]]; });
    subArr.sort(function(a,b) { return b[1]-a[1]; });
    var subTotal = subArr.reduce(function(s,x) { return s+x[1]; }, 0);

    spacer();
    // Title
    var titleRow = Array(NUM_COLS).fill('');
    titleRow[0] = itemName.toUpperCase() + ' - ' + label + ' - ' + channelLabel;
    pushRow(titleRow, true, 'FFC7CE');

    // Label
    var lRow = Array(NUM_COLS).fill('');
    lRow[0] = 'BOWLS - ' + label + ' (' + channelLabel + ') BY QUANTITY';
    lRow[3] = lRow[0];
    pushRow(lRow, false, null);

    // No Catering + first sub-table header
    var ncRow = Array(NUM_COLS).fill('');
    ncRow[0] = 'No Catering'; ncRow[3] = 'No Catering';
    if (hasSubTab && subArr.length > 0) {
      var sm0 = subArr[0][0]; var sq0 = subArr[0][1];
      var sc0 = getModCost_(sm0, modCostMap) || 0;
      ncRow[7] = sm0; ncRow[8] = sq0; ncRow[10] = sc0;
      ncRow[11] = subTotal ? sq0/subTotal : '';
      ncRow[12] = subTotal ? (sq0/subTotal)*sc0 : '';
    }
    pushRow(ncRow, false, null);

    // Header row (red) + second sub-table entry
    var hdrRow = Array(NUM_COLS).fill('');
    hdrRow[0] = 'modifier'; hdrRow[1] = 'SUM of Qty'; hdrRow[3] = 'Cost';
    if (hasSubTab && subArr.length > 1) {
      var sm1 = subArr[1][0]; var sq1 = subArr[1][1];
      var sc1 = getModCost_(sm1, modCostMap) || 0;
      hdrRow[7] = sm1; hdrRow[8] = sq1; hdrRow[10] = sc1;
      hdrRow[11] = subTotal ? sq1/subTotal : '';
      hdrRow[12] = subTotal ? (sq1/subTotal)*sc1 : '';
    }
    pushRow(hdrRow, true, 'C00000');

    // Data rows
    var modArr = Object.keys(mods).map(function(m) { return [m, mods[m]]; });
    modArr.sort(function(a,b) { return b[1]-a[1]; });
    var nonSkipTotal = modArr.reduce(function(s,x) { return s + (x[0].toLowerCase().indexOf('skip')!==0 ? x[1] : 0); }, 0);
    var secCostTotal = 0; var secQty = 0;

    modArr.forEach(function(pair, idx) {
      var modName = pair[0]; var qty = pair[1];
      var isSkip = modName.toLowerCase().indexOf('skip') === 0;
      var isComposite = (modName === '1/2 and 1/2 Base' || modName === '1/2 and 1/2 Mains' ||
                         modName === '1/2 and 1/2 Grains' || modName === '1/2 and 1/2 Greens');
      var unitCost = null; var rowTotal = null;
      if (isSkip) { unitCost = 0; rowTotal = 0; }
      else if (isComposite) { unitCost = subWeightedAvg_(sub, modCostMap); rowTotal = unitCost * qty; }
      else { unitCost = getModCost_(modName, modCostMap); rowTotal = (unitCost !== null) ? unitCost * qty : null; }
      if (rowTotal !== null) secCostTotal += rowTotal;
      if (!isSkip) secQty += qty;

      var rowPct  = (!isSkip && nonSkipTotal) ? qty/nonSkipTotal : '';
      var wtdCost = (rowPct !== '' && unitCost !== null) ? rowPct * unitCost : '';

      var dr = Array(NUM_COLS).fill('');
      dr[0] = modName; dr[1] = qty;
      dr[3] = (unitCost !== null) ? unitCost : '';
      if (baseCost === 0 && !hasSubTab) {
        dr[4] = rowPct !== '' ? rowPct : '';
        dr[5] = wtdCost !== '' ? wtdCost : '';
      } else {
        dr[4] = (rowTotal !== null) ? rowTotal : '';
      }

      // Sub-table (entries 2+, since 0=ncRow, 1=hdrRow consumed first two)
      var si = idx + 2;
      if (hasSubTab && si < subArr.length) {
        var sm = subArr[si][0]; var sq = subArr[si][1];
        var sc = getModCost_(sm, modCostMap) || 0;
        dr[7] = sm; dr[8] = sq; dr[10] = sc;
        dr[11] = subTotal ? sq/subTotal : '';
        dr[12] = subTotal ? (sq/subTotal)*sc : '';
      }
      pushRow(dr, false, null);
    });

    // Overflow sub-table entries
    for (var oi = modArr.length + 2; oi < subArr.length; oi++) {
      var xm = subArr[oi][0]; var xq = subArr[oi][1]; var xc = getModCost_(xm, modCostMap) || 0;
      var xRow = Array(NUM_COLS).fill('');
      xRow[7] = xm; xRow[8] = xq; xRow[10] = xc;
      xRow[11] = subTotal ? xq/subTotal : '';
      xRow[12] = subTotal ? (xq/subTotal)*xc : '';
      pushRow(xRow, false, null);
    }

    // Grand Total
    var gtRow = Array(NUM_COLS).fill('');
    gtRow[0] = 'Grand Total'; gtRow[1] = secQty;
    if (baseCost === 0 && !hasSubTab) {
      gtRow[5] = secQty ? secCostTotal/secQty : 0;
    } else {
      gtRow[4] = secCostTotal;
    }
    if (hasSubTab) { gtRow[7] = 'Grand Total'; gtRow[8] = subTotal; gtRow[12] = subWeightedAvg_(sub, modCostMap); }
    pushRow(gtRow, true, 'FFEB9C');

    if (!exclCost) totalModCost += secCostTotal;
  });

  // Final cost summary
  if (baseCost > 0) {
    var totalQtyForAvg = toRender.reduce(function(s, sec) {
      if (sec.key === 'Main' || sec.key === 'Base' || sec.key === '1/2 Base' || sec.key === 'Kids - Main') {
        return s + Object.keys(sec.mods).reduce(function(ss, m) {
          return ss + (m.toLowerCase().indexOf('skip')===0 ? 0 : sec.mods[m]);
        }, 0);
      }
      return s;
    }, 0) || 1;

    spacer();
    function sumRow_(lbl, val) {
      var r = Array(NUM_COLS).fill(''); r[0] = lbl; r[4] = val;
      return r;
    }
    pushRow(sumRow_('AVG COST OF ' + itemName.toUpperCase() + ' (' + channelLabel + ')', baseCost), true, 'FFEB9C');
    pushRow(sumRow_('TOTAL MODIFIER COST', totalModCost), false, 'FFEB9C');
    pushRow(sumRow_('TOTAL AVG COST', baseCost * totalQtyForAvg), false, 'FFEB9C');
    pushRow(sumRow_('MODIFIER + AVG COST', totalModCost + baseCost * totalQtyForAvg), false, 'FFEB9C');
    var finalAvg = baseCost + (totalQtyForAvg ? totalModCost / totalQtyForAvg : 0);
    pushRow(sumRow_('FINAL AVG COST WITH MODIFIER', finalAvg), true, 'FF0000');
  }

  // Write to sheet in one batch
  if (!allRows.length) return;
  ws.getRange(1, 1, allRows.length, NUM_COLS).setValues(allRows);

  // Apply styles
  for (var ri = 0; ri < allStyles.length; ri++) {
    var s = allStyles[ri];
    var cell = ws.getRange(ri + 1, 1, 1, NUM_COLS);
    if (s.bg) cell.setBackground('#' + s.bg);
    if (s.bold) cell.setFontWeight('bold');
    if (s.bg === 'C00000') cell.setFontColor('#FFFFFF');
    if (s.bg === 'FF0000') cell.setFontColor('#FFFFFF');
    cell.setFontFamily('Arial').setFontSize(9);
  }

  // Format dollar columns
  ws.getRange(1, 4,  allRows.length, 1).setNumberFormat('"$"#,##0.0000');
  ws.getRange(1, 5,  allRows.length, 1).setNumberFormat('"$"#,##0.0000');
  ws.getRange(1, 6,  allRows.length, 1).setNumberFormat('"$"#,##0.0000');
  ws.getRange(1, 11, allRows.length, 1).setNumberFormat('"$"#,##0.0000');
  ws.getRange(1, 12, allRows.length, 1).setNumberFormat('0.0000%');
  ws.getRange(1, 13, allRows.length, 1).setNumberFormat('"$"#,##0.0000');

  // Column widths
  [[1,240],[2,80],[3,20],[4,90],[5,100],[6,20],[7,20],[8,240],[9,80],[10,20],[11,90],[12,90],[13,110]]
    .forEach(function(cw) { ws.setColumnWidth(cw[0], cw[1]); });

  ws.setTabColor('#FF69B4');
}

function stepBuildChannelMasters() {
  var ss     = SpreadsheetApp.getActiveSpreadsheet();
  var cmData = getSheet_(SH.CLEAN_MB).getDataRange().getValues();

  // Build pink cost map: sheet name -> FINAL AVG COST WITH MODIFIER
  var pinkCostMap = buildPinkCostMap_(ss);

  var r365CostMap = buildR365CostMap_();

  buildOneMaster_(ss, cmData, 'IN HOUSE',  SH.MASTER_IN, 'IN-HOUSE', false, pinkCostMap, r365CostMap);

  toast_('Channel Masters built: IN, LO, 3P');
}

function buildR365CostMap_() {
  // Returns { 'CHANNEL||itemName': avgCost } from analytics.r365_item_cost
  // Also adds entries from analytics.r365_modifier_cost (MI rows) as fallback
  // for items like Strawberry Lassi, That Fire Hot Sauce that appear in modifier
  // cost sheet but not in item cost sheet.
  var MENU_TO_CH = {
    'DELIVERY':'DELIVERY','3PD OPEN MARKUP':'DELIVERY',
    'APP':'LOYALTY','FOOD - IN HOUSE':'IN HOUSE','DRINKS - IN HOUSE':'IN HOUSE',
  };
  var icData = getDataMap_(SH.R365_ITEM);
  var icH    = icData.headers;
  var map    = {};

  // 1. From r365_item_cost
  icData.rows.forEach(function(row) {
    var menu  = String(row[icH.indexOf('menu')]              || '').trim();
    var iname = String(row[icH.indexOf('item_name')]         || '').trim();
    var iupd  = String(row[icH.indexOf('item_name_updated')] || '').trim();
    var cost  = row[icH.indexOf('avg_cost')];
    var ch    = MENU_TO_CH[menu] || menu;
    if (typeof cost === 'number' && cost > 0) {
      if (iname) map[ch + '||' + iname] = cost;
      if (iupd && iupd !== iname) map[ch + '||' + iupd] = cost;
    }
  });

  // 2. From r365_modifier_cost MI rows — for standalone items not in r365_item_cost
  //    (e.g. Strawberry Lassi, That Fire Hot Sauce).
  //    Key: 'ANY||cleanName' — used as a universal fallback in getR365ItemCost_.
  var mcData = getDataMap_(SH.R365_MOD);
  var mcH    = mcData.headers;
  mcData.rows.forEach(function(row) {
    var rname = String(row[mcH.indexOf('recipe_name')]    || '').trim();
    var cname = String(row[mcH.indexOf('clean_name')]     || '').trim();
    var cost  = row[mcH.indexOf('cost_per_portion')];
    if (rname.indexOf('MI ') !== 0 || !cname) return;
    if (typeof cost !== 'number' || cost <= 0) return;
    var key = 'ANY||' + cname;
    if (!map[key] || cost > map[key]) map[key] = cost;
    // Also try with lowercased name for fuzzy match
    var keyLow = 'ANY_LOW||' + cname.toLowerCase().trim();
    if (!map[keyLow] || cost > map[keyLow]) map[keyLow] = cost;
  });

  return map;
}

function buildPinkCostMap_(ss) {
  // Reads each pink sheet and extracts the avg cost with modifiers.
  // For items with a base cost: reads FINAL AVG COST WITH MODIFIER from col E.
  // For zero-baseCost items (Sides, Homemade Juice, Handcrafted Juice etc.):
  //   reads Grand Total row col F = weighted avg of (unit cost × mix %) across modifiers.
  var pinkCosts = {};
  var MAX_SCAN = 200;
  var SKIP_PATTERN = /menu item master|clean |raw |lookup|step|bikky|validation|blended|overall|pivot|helper|tables|rough|average costs|public\.|analytics\./i;
  ss.getSheets().forEach(function(ws) {
    var name = ws.getName();
    if (SKIP_PATTERN.test(name)) return;
    var maxRow = Math.min(ws.getLastRow(), MAX_SCAN);
    if (maxRow < 2) return;
    var colA = ws.getRange(1, 1, maxRow, 6).getValues();  // read cols A-F in one call

    // 1. Look for FINAL AVG COST WITH MODIFIER in col A, value in col E (index 4)
    for (var i = maxRow - 1; i >= 0; i--) {
      var label = String(colA[i][0] || '').trim().toUpperCase();
      if (label.indexOf('FINAL AVG COST WITH MODIFIER') !== -1) {
        var val = colA[i][4];
        if (typeof val === 'number' && val > 0) { pinkCosts[name] = val; break; }
      }
    }

    // 2. Fallback for zero-baseCost items:
    //    Grand Total row has weighted avg cost in col F (index 5).
    if (!pinkCosts[name]) {
      for (var j = 0; j < maxRow; j++) {
        var lbl = String(colA[j][0] || '').trim().toLowerCase();
        if (lbl === 'grand total') {
          var v = colA[j][5];  // col F
          if (typeof v === 'number' && v > 0) { pinkCosts[name] = v; break; }
        }
      }
    }
  });
  return pinkCosts;
}

function getPinkCost_(itemName, channel, pinkCostMap) {
  // Returns FINAL AVG COST WITH MODIFIER from pink sheet.
  // Returns null if no pink sheet exists for this item
  // (callers should fall back to r365_item_cost for standalone items like drinks/sweets).
  var isIH = (channel === 'IN HOUSE');
  var base = itemName.endsWith(' - In House') ? itemName.slice(0, -11) : itemName;
  if (isIH) {
    var ihKey = (base + '(in house)').length > 31 ? (base + '(in house)').slice(0, 31) : base + '(in house)';
    if (pinkCostMap[ihKey] !== undefined) return pinkCostMap[ihKey];
    if (pinkCostMap[base] !== undefined) return pinkCostMap[base];
  } else {
    var onKey = base.length > 31 ? base.slice(0, 31) : base;
    if (pinkCostMap[onKey] !== undefined) return pinkCostMap[onKey];
  }
  return null;
}

function getR365ItemCost_(itemName, channel, r365CostMap) {
  // Fallback cost for items without pink sheets (drinks, sweets, retail).
  // Priority: r365_item_cost (channel-specific) → r365_item_cost (any channel)
  //           → r365_modifier_cost MI rows (ANY key) → name-variant matching.
  var isIH = (channel === 'IN HOUSE');
  var base  = itemName.endsWith(' - In House') ? itemName.slice(0, -11) : itemName;

  // 1. r365_item_cost — IH-specific variants
  if (isIH) {
    var IH_VARIANTS = {
      'Kids Meal': 'Kids BYO - In House', 'BYO Indian Burrito': 'Burrito - In House',
      'BYO Grain Bowl': 'BYO Grain Bowl - In House', 'BYO Salad Bowl': 'BYO Salad Bowl - In House',
      'BYO Greens + Grains Bowl': 'BYO Greens + Grains Bowl - In House',
    };
    var varKey = IH_VARIANTS[base];
    if (varKey && r365CostMap['IN HOUSE||' + varKey] !== undefined) return r365CostMap['IN HOUSE||' + varKey];
    if (r365CostMap['IN HOUSE||' + base + ' - In House'] !== undefined) return r365CostMap['IN HOUSE||' + base + ' - In House'];
    if (r365CostMap['IN HOUSE||' + base] !== undefined) return r365CostMap['IN HOUSE||' + base];
  }

  // 2. r365_item_cost — any channel
  if (r365CostMap['DELIVERY||' + base] !== undefined) return r365CostMap['DELIVERY||' + base];
  if (r365CostMap['LOYALTY||'  + base] !== undefined) return r365CostMap['LOYALTY||'  + base];
  if (r365CostMap['IN HOUSE||' + base] !== undefined) return r365CostMap['IN HOUSE||' + base];

  // 3. r365_modifier_cost MI rows (ANY key) — exact match
  if (r365CostMap['ANY||' + base] !== undefined) return r365CostMap['ANY||' + base];

  // 4. Lowercased match (handles 'That Fire Hot Sauce' -> MI entry)
  var baseLow = base.toLowerCase().trim();
  if (r365CostMap['ANY_LOW||' + baseLow] !== undefined) return r365CostMap['ANY_LOW||' + baseLow];

  // 5. Strip ' - ' → ' ' (handles 'LaCroix - Lime' → 'LaCroix Lime' → 'lacroix lime')
  var baseLow2 = baseLow.replace(/ - /g, ' ');
  if (r365CostMap['ANY_LOW||' + baseLow2] !== undefined) return r365CostMap['ANY_LOW||' + baseLow2];

  return null;
}

function buildOneMaster_(ss, cmData, channel, sheetName, revCenter, is3PD, pinkCostMap, r365CostMap) {
  // Aggregate qty and weighted price from Clean MB
  var items = {};
  var hdrs = cmData[0];
  var C = {
    menu: hdrs.indexOf('Menu'),
    iname: hdrs.indexOf('Item name_updated'),
    grp:   hdrs.indexOf('Menu Group'),
    price: hdrs.indexOf('Avg Price'),
    qty:   hdrs.indexOf('Quantity'),
  };
  for (var r = 1; r < cmData.length; r++) {
    if (String(cmData[r][C.menu] || '') !== channel) continue;
    var name  = String(cmData[r][C.iname] || '').trim();
    var grp   = String(cmData[r][C.grp]   || '').trim();
    var price = Number(cmData[r][C.price]) || 0;
    var qty   = Number(cmData[r][C.qty])   || 0;
    if (!name || qty === 0) continue;
    if (!items[name]) items[name] = { qty: 0, pw: 0, grp: grp };
    items[name].qty += qty;
    items[name].pw  += price * qty;
    if (!items[name].grp && grp) items[name].grp = grp;
  }

  var names = Object.keys(items).sort();
  var totQty = names.reduce(function(s, n) { return s + items[n].qty; }, 0);

  // First pass: compute costs and NS for thresholds
  var itemCalc = {};
  names.forEach(function(n) {
    var d   = items[n];
    var rp  = d.qty ? d.pw / d.qty : 0;
    var ap  = is3PD ? rp * RATE_3PD_PRICE : rp;
    // For 3PD: use the online pink sheet cost * 1.18
    var lookupChannel = is3PD ? 'DELIVERY' : channel;
    var pc = getPinkCost_(n, lookupChannel, pinkCostMap);
    var ac;
    if (pc !== null) {
      // Pink sheet cost found — this is the FINAL AVG COST WITH MODIFIER (online base)
      ac = is3PD ? pc * RATE_3PD_COST : pc;
    } else {
      // No pink sheet (drinks, sweets, retail, alc) — use r365_item_cost directly
      var fallback = getR365ItemCost_(n, channel, r365CostMap);
      ac = fallback !== null ? (is3PD ? fallback * RATE_3PD_COST : fallback) : 0;
    }
    var ns  = ap * d.qty;
    var tc  = ac * d.qty;
    itemCalc[n] = { ap: ap, ac: ac, qty: d.qty, ns: ns, tc: tc, grp: d.grp };
  });

  var totNS  = names.reduce(function(s, n) { return s + itemCalc[n].ns; }, 0);
  var totTC  = names.reduce(function(s, n) { return s + itemCalc[n].tc; }, 0);
  var medM   = totNS ? (totNS - totTC) / totNS : 0;
  var medMM  = names.length ? (1 / names.length) * 0.7 : 0;

  // Category NS totals
  var catNS = {}; var subNS = {};
  names.forEach(function(n) {
    var cat = getCat_(n, itemCalc[n].grp);
    var sub = getSubCat_(n, itemCalc[n].grp);
    catNS[cat] = (catNS[cat] || 0) + itemCalc[n].ns;
    subNS[sub] = (subNS[sub] || 0) + itemCalc[n].ns;
  });

  var hasBikky = (channel === 'IN HOUSE' || channel === 'LOYALTY');
  var headers = [
    'Item Name','Avg Price','Avg Cost With Modifiers','Margin',' Quantity',
    'Total Cost ','Net Sales','Total Margin','COGS%','% Margin',
    '% Menu Mix','Margin 2','Menu Mix','Menu Engineering - Final ',
    'Revenue Center','Category','Sub Category','Sls %','Sls % Category','Sls % Sub Category',
  ];
  if (hasBikky) headers.push('90 day Item Return Rate', '90 day Item Reorder Rate');

  var rows = [];
  names.forEach(function(n) {
    var d   = itemCalc[n];
    var mgn = d.ap - d.ac;
    var tmgn= d.ns - d.tc;
    var cogs= d.ns  ? d.tc / d.ns  : 0;
    var mpct= d.ap  ? mgn  / d.ap  : 0;
    var mmpct=totQty? d.qty/ totQty: 0;
    var spct= totNS ? d.ns / totNS : 0;
    var cat = getCat_(n, d.grp);
    var sub = getSubCat_(n, d.grp);
    var scat= catNS[cat] ? d.ns / catNS[cat] : 0;
    var ssub= subNS[sub] ? d.ns / subNS[sub] : 0;
    var mh  = mpct  > medM;
    var mmh = mmpct > medMM;
    var row = [n, d.ap, d.ac, mgn, d.qty, d.tc, d.ns, tmgn, cogs, mpct,
               mmpct, mh ? 'High' : 'Low', mmh ? 'High' : 'Low', menuEng_(mh, mmh),
               revCenter, cat, sub, spct, scat, ssub];
    if (hasBikky) row.push('', '');  // Bikky populated in Step 9
    rows.push(row);
  });

  var nc = headers.length;
  var gtRow = new Array(nc).fill('');
  gtRow[0]  = 'GRAND TOTAL'; gtRow[4] = totQty; gtRow[6] = totNS;
  gtRow[14] = revCenter; gtRow[17] = 1; gtRow[18] = 1; gtRow[19] = 1;

  var mgtRow = new Array(nc).fill('');
  mgtRow[0]  = 'Avg Margin Threshold (SUM Margin$ / SUM Sales$)';
  mgtRow[9]  = medM;

  var mmtRow = new Array(nc).fill('');
  mmtRow[0]  = 'Menu Mix Threshold (1/n × 0.7)';
  mmtRow[10] = medMM;

  var ws = clearAndCreate_(sheetName);
  writeSheetData_(ws, headers, rows, gtRow, [mgtRow, mmtRow]);

  // Format columns
  var dr = rows.length;
  var base2 = dr + 2;  // data starts row 2, gt at dr+2, thresholds at dr+3 and dr+4
  [[2,'"$"#,##0.00'],[3,'"$"#,##0.00'],[4,'"$"#,##0.00'],
   [6,'"$"#,##0.00'],[7,'"$"#,##0.00'],[8,'"$"#,##0.00'],
   [9,'0.00%'],[10,'0.00%'],[11,'0.00%'],[18,'0.00%'],[19,'0.00%'],[20,'0.00%']
  ].forEach(function(cf) {
    ws.getRange(2, cf[0], dr, 1).setNumberFormat(cf[1]);
  });
  // Threshold rows formatting
  ws.getRange(base2 + 1, 10).setNumberFormat('0.00%');
  ws.getRange(base2 + 2, 11).setNumberFormat('0.000%');

  ws.setColumnWidth(1, 240); ws.setColumnWidth(14, 120); ws.setColumnWidth(15, 100);
  ws.setColumnWidth(16, 100); ws.setColumnWidth(17, 100);
}

// STEP 7 — BUILD BLENDED MASTER
// ============================================================
function stepBuildBlendedMaster() {
  var items = {};
  [SH.MASTER_IN, SH.MASTER_LO, SH.MASTER_3P].forEach(function(shName) {
    var ws = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(shName);
    if (!ws) return;
    var data = ws.getDataRange().getValues();
    for (var r = 1; r < data.length; r++) {
      var n = String(data[r][0] || '').trim();
      if (!n || n === 'GRAND TOTAL' || n.indexOf('Avg Margin') === 0 || n.indexOf('Menu Mix') === 0) continue;
      var base = n.endsWith(' - In House') ? n.slice(0, -11) : n;
      var ap = Number(data[r][1]) || 0;
      var ac = Number(data[r][2]) || 0;
      var qty= Number(data[r][4]) || 0;
      var ns = Number(data[r][6]) || 0;
      var cat= String(data[r][15] || '').trim();
      var sub= String(data[r][16] || '').trim();
      if (!items[base]) items[base] = { qty:0, ns:0, pw:0, cw:0, cat:cat, sub:sub };
      items[base].qty += qty; items[base].ns  += ns;
      items[base].pw  += ap * qty; items[base].cw  += ac * qty;
    }
  });

  var names = Object.keys(items).sort();
  var totQty = names.reduce(function(s,n) { return s+items[n].qty; }, 0);
  var totNS  = names.reduce(function(s,n) { return s+items[n].ns;  }, 0);
  var totTC  = names.reduce(function(s,n) { return s+items[n].cw;  }, 0);
  var medM   = totNS ? (totNS-totTC)/totNS : 0;
  var medMM  = names.length ? (1/names.length)*0.7 : 0;

  var catNS = {}; var subNS = {};
  names.forEach(function(n) {
    catNS[items[n].cat] = (catNS[items[n].cat]||0) + items[n].ns;
    subNS[items[n].sub] = (subNS[items[n].sub]||0) + items[n].ns;
  });

  var headers = [
    'Item  Name','Avg Price','Avg Cost With Modifiers','Margin',' Quantity',
    'Total Cost ','Net Sales','Total Margin','COGS%','% Margin',
    '% Menu Mix','Margin 2','Menu Mix','Menu Engineering - Final ',
    'Category','Sub Category','Sls %','Sls % Category','Sls % Sub Category',
  ];

  var rows = [];
  names.forEach(function(n) {
    var d = items[n];
    var ap = d.qty ? d.pw/d.qty : 0; var ac = d.qty ? d.cw/d.qty : 0;
    var mgn= ap-ac; var tc=ac*d.qty; var tmgn=d.ns-tc;
    var cogs=d.ns?tc/d.ns:0; var mpct=ap?mgn/ap:0;
    var mmpct=totQty?d.qty/totQty:0; var spct=totNS?d.ns/totNS:0;
    var scat=catNS[d.cat]?d.ns/catNS[d.cat]:0;
    var ssub=subNS[d.sub]?d.ns/subNS[d.sub]:0;
    var mh=mpct>medM; var mmh=mmpct>medMM;
    rows.push([n,ap,ac,mgn,d.qty,tc,d.ns,tmgn,cogs,mpct,
               mmpct,mh?'High':'Low',mmh?'High':'Low',menuEng_(mh,mmh),
               d.cat,d.sub,spct,scat,ssub]);
  });

  var nc = headers.length;
  var mgtRow = new Array(nc).fill(''); mgtRow[0]='Avg Margin Threshold (SUM Margin$ / SUM Sales$)'; mgtRow[9]=medM;
  var mmtRow = new Array(nc).fill(''); mmtRow[0]='Menu Mix Threshold (1/n × 0.7)'; mmtRow[10]=medMM;

  var ws = clearAndCreate_(SH.MASTER_BL);
  writeSheetData_(ws, headers, rows, null, [mgtRow, mmtRow]);

  var dr = rows.length;
  [[2,'"$"#,##0.00'],[3,'"$"#,##0.00'],[4,'"$"#,##0.00'],
   [6,'"$"#,##0.00'],[7,'"$"#,##0.00'],[8,'"$"#,##0.00'],
   [9,'0.00%'],[10,'0.00%'],[11,'0.00%'],[17,'0.00%'],[18,'0.00%'],[19,'0.00%'],
  ].forEach(function(cf) { ws.getRange(2, cf[0], dr, 1).setNumberFormat(cf[1]); });

  ws.setColumnWidth(1, 240);
  toast_('Blended Master: ' + names.length + ' items');
}

// STEP 8 — BUILD OVERALL MASTER (stacked all channels)

function stepBuildOverallMaster() {
  var allRows = [];
  var masterSheets = [
    { name: SH.MASTER_IN, rev: 'IN-HOUSE' },
    { name: SH.MASTER_LO, rev: 'LOYALTY' },
    { name: SH.MASTER_3P, rev: '3PD' },

  masterSheets.forEach(function(ms) {
    var ws = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(ms.name);

                     tc:data[r][5], ns:data[r][6], tmgn:data[r][7], cogs:data[r][8],

    }
  });

  var totQty = allRows.reduce(function(s,r) { return s+(Number(r.qty)||0); }, 0);

    'Total Cost ','Net Sales','Total Margin','COGS%','% Margin',
    '% Menu Mix','Margin 2','Menu Mix','Menu Engineering - Final',

  var rows = [];

    var qty=Number(r.qty)||0; var ns=Number(r.ns)||0;

    rows.push([r.item,r.ap,r.ac,r.mgn,qty,r.tc,ns,r.tmgn,r.cogs,r.mpct,
               mmpct,r.mh,r.mmh,r.eng,r.rev,r.cat,r.sub,spct,scat]);

  var ws = clearAndCreate_(SH.MASTER_OV);
  writeSheetData_(ws, headers, rows, null, null);

  var dr = rows.length;
  [[2,'"$"#,##0.00'],[3,'"$"#,##0.00'],[4,'"$"#,##0.00'],

   [9,'0.00%'],[10,'0.00%'],[11,'0.00%'],[18,'0.00%'],[19,'0.00%'],
  ].forEach(function(cf) { ws.getRange(2, cf[0], dr, 1).setNumberFormat(cf[1]); });

  ws.setColumnWidth(1, 240); ws.setColumnWidth(14, 100);
  toast_('Overall Master: ' + allRows.length + ' item-channel rows');
}

// STEP 9 — POPULATE BIKKY RATES
// Reads fact_bikky_instore -> IH master
//         fact_bikky_3pd_loyalty -> LO and 3P masters
// ============================================================
function stepPopulateBikky() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  var BIKKY_ALIASES = {
    'Grain Bowl':           'BYO Grain Bowl',
    'Salad Bowl':           'BYO Salad Bowl',
    'Greens + Grains Bowl': 'BYO Greens + Grains Bowl',
    'Burrito':              'BYO Indian Burrito',
    'Cauliflower + Quinoa': 'Spiced Cauli + Quinoa Bowl',
    'Kids BYO':             'Kids Meal',
  };

  function buildBikkyMap(shName) {
    var data = getDataMap_(shName);
    var H = data.headers;
    var map = {};
    data.rows.forEach(function(row) {
      var rawName = String(row[H.indexOf('item_name')] || '').trim();
      var name    = BIKKY_ALIASES[rawName] || rawName;
      var ret     = row[H.indexOf('return_rate')];
      var reo     = row[H.indexOf('reorder_rate')];
      if (name && typeof ret === 'number') {
        map[name] = { ret: ret, reo: (typeof reo === 'number' ? reo : '') };
      }
    });
    return map;
  }

  var ihMap = buildBikkyMap(SH.BIK_IH);
  var tpMap = buildBikkyMap(SH.BIK_3P);

  function applyBikky(shName, bikkyMap) {
    var ws = ss.getSheetByName(shName);
    if (!ws) return 0;
    var data = ws.getDataRange().getValues();
    var updated = 0;
    for (var r = 1; r < data.length; r++) {
      var name = String(data[r][0] || '').trim();
      if (!name || name === 'GRAND TOTAL' || name.indexOf('Avg') === 0 || name.indexOf('Menu Mix') === 0) continue;
      var found = bikkyMap[name];
      if (found) {
        ws.getRange(r + 1, 21).setValue(found.ret);
        ws.getRange(r + 1, 22).setValue(found.reo !== '' ? found.reo : '');
        updated++;
      }
    }
    // Format Bikky columns
    if (ws.getLastRow() > 1) {
      ws.getRange(2, 21, ws.getLastRow() - 1, 2).setNumberFormat('0.00%');
    }
    return updated;
  }

  var ihUpd = applyBikky(SH.MASTER_IN, ihMap);
  var loUpd = applyBikky(SH.MASTER_LO, tpMap);
  var tpUpd = applyBikky(SH.MASTER_3P, tpMap);

  toast_('Bikky: IH ' + ihUpd + ' | LO ' + loUpd + ' | 3PD ' + tpUpd + ' rates updated');
  SpreadsheetApp.getUi().alert('Bikky Complete',
    'In House:  ' + ihUpd + ' items updated\n' +
    'Loyalty:   ' + loUpd + ' items updated\n' +
    '3PD:       ' + tpUpd + ' items updated\n\n' +
    'Items not in Bikky retain blank values.',
    SpreadsheetApp.getUi().ButtonSet.OK);
}
